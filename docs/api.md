## Quick Start (pure library)

```ts
import { ConnectorManager } from "@naculus/connect-core";
import { createWalletConnectConnector } from "@naculus/connector-walletconnect";
import { eip6963Connector } from "@naculus/connector-evm-injected";

const manager = new ConnectorManager();
manager.register("walletconnect", createWalletConnectConnector({
  projectId: "your-project-id",
  metadata: { name: "My App", description: "...", url: "...", icons: [] },
}));
manager.register("eip6963", eip6963Connector);

const session = await manager.connect("walletconnect");
```
